Database Schema Design
Core Tables
sql
-- Personalized Plan Main Table
PersonalizedPlan(
    id = UUID (Primary Key)
    user = ForeignKey(UserRegistration)
    plan_type = CharField (role/language)
    target_role/language = CharField
    current_week = IntegerField (default=1)
    current_day = IntegerField (default=1)
    status = CharField (active/paused/completed/failed)
    created_at = DateTimeField
    updated_at = DateTimeField
)
 
-- Weekly Plans
WeeklyPlan(
    id = UUID (Primary Key)
    personalized_plan = ForeignKey(PersonalizedPlan)
    week_number = IntegerField
    skill_focus = CharField
    topics = JSONField (daily topics)
    status = CharField (locked/in_progress/completed/failed)
    extra_days_added = IntegerField (default=0)
    weak_topics_focus = JSONField (null=True)
    created_at = DateTimeField
)
 
-- Daily Tasks
DailyTask(
    id = UUID (Primary Key)
    weekly_plan = ForeignKey(WeeklyPlan)
    day_number = IntegerField
    topic = CharField
    description = TextField
    status = CharField (locked/unlocked/completed)
    completed_at = DateTimeField (null=True)
    xp_reward = IntegerField (default=15)
)
 
-- User Progress Tracking
PlanProgress(
    id = UUID (Primary Key)
    user = ForeignKey(UserRegistration)
    personalized_plan = ForeignKey(PersonalizedPlan)
    current_week = IntegerField
    current_day = IntegerField
    total_days_completed = IntegerField
    weekly_tests_passed = IntegerField
    weekly_tests_failed = IntegerField
    last_activity = DateTimeField
)
 
-- Assessment Results
AssessmentResult(
    id = UUID (Primary Key)
    user = ForeignKey(UserRegistration)
    personalized_plan = ForeignKey(PersonalizedPlan)
    test_type = CharField (initial/weekly/final/correction)
    score = IntegerField
    total_questions = IntegerField
    weak_areas = JSONField
    created_at = DateTimeField
)
 
-- Weak Topic Diagnosis
WeakTopicDiagnosis(
    id = UUID (Primary Key)
    assessment_result = ForeignKey(AssessmentResult)
    weak_topics = JSONField (topic names and scores)
    severity_level = CharField (low/medium/high)
    created_at = DateTimeField
)
 
-- Correction Plans (2 weeks)
CorrectionPlan(
    id = UUID (Primary Key)
    user = ForeignKey(UserRegistration)
    personalized_plan = ForeignKey(PersonalizedPlan)
    duration_weeks = IntegerField (default=2)
    all_topics = JSONField (comprehensive topic list)
    status = CharField (active/completed/failed)
    created_at = DateTimeField
)
 
-- Final Retest Results
FinalRetest(
    id = UUID (Primary Key)
    correction_plan = ForeignKey(CorrectionPlan)
    score = IntegerField
    total_questions = IntegerField (60)
    career_recommendation = CharField
    readiness_level = CharField
    created_at = DateTimeField
)
 
-- XP System
UserXP(
    id = UUID (Primary Key)
    user = ForeignKey(UserRegistration, OneToOne)
    total_xp = IntegerField (default=0)
    current_level = IntegerField (default=1)
    xp_to_next_level = IntegerField (100)
    level_progress = IntegerField (0-100)
    last_updated = DateTimeField
)
 
-- XP Reward Transactions
XPReward(
    id = UUID (Primary Key)
    user_xp = ForeignKey(UserXP)
    reward_type = CharField (daily_task/weekly_test/streak/recovery/speed/consistency)
    xp_amount = IntegerField
    reason = CharField
    related_object_id = UUIDField (task/assessment ID)
    created_at = DateTimeField
)
 
-- Streak Tracking
UserStreak(
    id = UUID (Primary Key)
    user = ForeignKey(UserRegistration, OneToOne)
    current_streak = IntegerField (default=0)
    longest_streak = IntegerField (default=0)
    last_activity_date = DateField (null=True)
    total_days_active = IntegerField (default=0)
    created_at = DateTimeField
)
 
-- Daily Activity Log
DailyActivity(
    id = UUID (Primary Key)
    user = ForeignKey(UserRegistration)
    date = DateField
    week_number = IntegerField
    day_number = IntegerField
    is_active = BooleanField (default=False)
    tasks_completed = IntegerField (default=0)
    xp_earned = IntegerField (default=0)
    created_at = DateTimeField
)
 
-- Resume Impact Tracking
ResumeImpact(
    id = UUID (Primary Key)
    user = ForeignKey(UserRegistration)
    role = CharField
    skill = CharField
    before_score = IntegerField
    after_score = IntegerField
    improvement_percentage = FloatField
    created_at = DateTimeField
)
Relationships and Constraints
One user can have only one active PersonalizedPlan
PersonalizedPlan has 3-4 WeeklyPlans
Each WeeklyPlan has 7 DailyTasks
XP updates trigger level calculations automatically
All foreign key relationships with CASCADE delete
